class Solution {
public:
    int smallestRangeII(vector<int>& nums, int k) {
        sort(nums.begin(),nums.end());
        int size=nums.size();
        int ans=nums[size-1]-nums[0];
        for(int i=0;i<=size-2;i++)
        {
            int now_min=min(nums[0]+k,nums[i+1]-k);
            int now_max=max(nums[i]+k,nums[size-1]-k);
            ans=min(ans,now_max-now_min);
        } 
        return ans;
    }
};
