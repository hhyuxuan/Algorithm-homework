class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        int targetIndex = nums.size() - k;
        return select(nums, 0, nums.size() - 1, targetIndex);
    }

private:
    int select(vector<int>& nums, int left, int right, int k) {
        if (left == right) return nums[left];

        int pivotValue = getPivotValue(nums, left, right);
        pair<int, int> range = partition(nums, left, right, pivotValue);

        if (k >= range.first && k <= range.second) {
            return nums[k];
        } else if (k < range.first) {
            return select(nums, left, range.first - 1, k);
        } else {
            return select(nums, range.second + 1, right, k);
        }
    }

    int getPivotValue(vector<int>& nums, int left, int right) {
        int size = right - left + 1;
        if (size <= 5) return findMedianDirectly(nums, left, right);

        vector<int> medians;
        for (int i = left; i <= right; i += 5) {
            int subRight = min(i + 4, right);
            medians.push_back(findMedianDirectly(nums, i, subRight));
        }
        return select(medians, 0, medians.size() - 1, medians.size() / 2);
    }

    int findMedianDirectly(vector<int>& nums, int left, int right) {
        sort(nums.begin() + left, nums.begin() + right + 1);
        return nums[left + (right - left) / 2];
    }

    pair<int, int> partition(vector<int>& nums, int left, int right, int pivot) {
        int lt = left;
        int i = left;
        int gt = right;

        while (i <= gt) {
            if (nums[i] < pivot) {
                swap(nums[lt++], nums[i++]);
            } else if (nums[i] > pivot) {
                swap(nums[i], nums[gt--]);
            } else {
                i++;
            }
        }
        return {lt, gt};
    }
};
